说到关于迷宫课设

原文发表于2014-08-26

 

群里聊天，听说上一届C语言的课设是做个迷宫，我当然是不会啦，但从视觉效果上我写了这么个东西。喜感啊

疑问：

①SetConsoleOutputCP，是否跟那个啥Unicode的页有关系？

#include <stdio.h>
#include <stdlib.h>
#include <windows.h>

int i,j,k;

int main()
{
    SetConsoleOutputCP(437);

    for (k = 0; k < 500; k++)
    {
        if (k % 500 == 0)
            printf("\n");
        else
            for (i = 0; i < 10; i++)
            {   
                j = rand();
                if      (j > 0 && j <= 3000)
                    {}
                else if (j > 3000 && j <= 6000)
                    printf("%c",185);  
                else if (j > 6000 && j <= 9000)
                    printf("%c",186);  
                else if (j > 9000 && j <= 12000)
                    printf("%c",187);  
                else if (j > 12000 && j <= 15000)
                    printf("%c",188);  
                else if (j > 15000 && j <= 18000)
                    printf("%c",200);  
                else if (j > 18000 && j <= 21000)
                    printf("%c",201);  
                else if (j > 21000 && j <= 24000)
                    printf("%c",202);  
                else if (j > 27000 && j <= 29000)
                    printf("%c",203);  
                else 
                    printf("%c",204);   
            } 
    } 

    getchar();
    return 0;    
}
